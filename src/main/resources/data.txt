create table if not exists Category(
    id int auto_increment primary key,
    description varchar(255),
    acronym varchar(255)
);

create table if not exists ClassCategory(
    id int auto_increment primary key,
    description varchar(255),
    acronym varchar(255)
);

create table if not exists CategorizedItemComposition(
    id int auto_increment primary key,
    description varchar(255),
    acronym varchar(255),
    classCategory_id int,
    category_id int,
    foreign key (classCategory_id) references ClassCategory(id),
    foreign key (category_id) references Category(id)
);
